def runif(x, a, c, m, n):
    '''
    x: seed
    a: multiplicador
    c: incremento
    m: modulo
    n: quantidade de variáveis

    the best parameters: a = 48271, c = 0 if multiplicative or c = 1 if aditive , m = 2147483647.
    '''
    x = x
    x = (a * x + c) % m
    l_xp = []
    for i in range(n):
        x = (a * x + c) % m
        l_xp.append(x / m)
    return l_xp

def exponen(U, lambda__):
  U = np.array(U)
  return -1/lambda__ * np.log(U)

def geometric(U, p):
  U = np.array(U)
  return np.ceil(np.log(1 - U) / np.log(1-p))
